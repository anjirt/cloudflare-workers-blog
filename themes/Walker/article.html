import os

path = 'C:\Users\Yuyy\Desktop\ShortVideo'  # 要遍历的目录
for root, dirs, names in os.walk(path):
    for name in names:
        ext = os.path.splitext(name)[1]  # 获取后缀名
        if ext == '.mp4':
            fromdir = os.path.join(root, name)  # mp4文件原始地址
            moveto = os.path.join("C:\Users\Yuyy\Desktop\Video", name)
            #dirname 上一层目录  
            os.rename(fromdir, moveto)  # 移动文件
